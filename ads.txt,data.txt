    <script>
        // Cache Busting Utility
        function applyCacheBuster(url) {
            const cacheBusterParam = `v=${new Date().getTime()}`;
            return url.includes('?') ? `${url}&${cacheBusterParam}` : `${url}?${cacheBusterParam}`;
        }

        let appData = [];
        let adsData = [];
        let currentAdIndex = 0;
        let adInterval;
        let favorites = JSON.parse(localStorage.getItem('chakradham_favs')) || [];

        const uiStateContainer = document.getElementById('uiStateContainer');
        const cardContainer = document.getElementById('cardContainer');

        // Fetch Data Function
        async function fetchTextData(fileName, type) {
            const urlWithCacheBusting = applyCacheBuster(fileName);
            
            try {
                const response = await fetch(urlWithCacheBusting);
                if (!response.ok) {
                    throw new Error(`Failed to load ${fileName} (Status: ${response.status})`);
                }
                const text = await response.text();
                return parseTextData(text, type);
            } catch (error) {
                console.error("Fetch Error:", error);
                showError(`Data Connection Error <br><span class="text-xs text-slate-500 mt-2 block">${error.message}</span>`);
                return null;
            }
        }

        // Updated Parser for Vertical Key: Value Format
        function parseTextData(text, type) {
            const blocks = text.split('---').map(block => block.trim()).filter(block => block !== '');
            const parsedArray = [];

            blocks.forEach(block => {
                const lines = block.split('\n');
                const itemData = {};

                lines.forEach(line => {
                    const colonIndex = line.indexOf(':');
                    if (colonIndex !== -1) {
                        const key = line.substring(0, colonIndex).trim();
                        const value = line.substring(colonIndex + 1).trim();
                        itemData[key] = value;
                    }
                });

                if (type === 'cards' && Object.keys(itemData).length > 0) {
                    parsedArray.push({
                        id: itemData['ID'] || '',
                        title: itemData['Title'] || 'Untitled',
                        type: itemData['Type'] || 'Unknown',
                        date: itemData['Date'] || 'N/A',
                        link: itemData['Link'] || '#'
                    });
                } else if (type === 'ads' && Object.keys(itemData).length > 0) {
                    parsedArray.push({
                        img: itemData['Image'] || '',
                        link: itemData['Link'] || '#'
                    });
                }
            });

            return parsedArray;
        }

        // Error Message UI (Dark Theme)
        function showError(message) {
            uiStateContainer.innerHTML = `
                <div class="bg-slate-800/80 border border-red-500/30 p-8 rounded-2xl text-center max-w-md w-full shadow-lg">
                    <div class="w-16 h-16 bg-red-500/10 rounded-full flex items-center justify-center mx-auto mb-4">
                        <i class="fa-solid fa-triangle-exclamation text-3xl text-red-500"></i>
                    </div>
                    <p class="text-slate-300 font-medium">${message}</p>
                </div>`;
            uiStateContainer.style.display = 'flex';
        }

        // Ads Logic
        function startAds() {
            if (adsData.length === 0) return;
            
            document.getElementById('adContainer').classList.remove('hidden');
            const adImageEl = document.getElementById('adImage');
            const adLinkEl = document.getElementById('adLink');

            function changeAd() {
                const newImgSrc = applyCacheBuster(adsData[currentAdIndex].img);
                
                adImageEl.style.opacity = 0;
                setTimeout(() => {
                    adImageEl.src = newImgSrc;
                    adLinkEl.href = adsData[currentAdIndex].link;
                    adImageEl.style.opacity = 1;
                }, 400); // Slightly smoother fade for dark theme
                
                currentAdIndex = (currentAdIndex + 1) % adsData.length;
            }
            
            changeAd();
            adInterval = setInterval(changeAd, 5000);
        }

        // Render Cards UI (New Dark & Neon Style)
        function renderCards(filterText = "") {
            cardContainer.innerHTML = "";
            
            const filteredData = appData.filter(item => item.id.includes(filterText));

            if (filteredData.length === 0) {
                uiStateContainer.innerHTML = `
                    <div class="text-center py-16">
                        <div class="w-20 h-20 bg-slate-800 rounded-full flex items-center justify-center mx-auto mb-4 border border-slate-700">
                            <i class="fa-solid fa-ghost text-4xl text-slate-600"></i>
                        </div>
                        <p class="text-slate-400 font-medium text-lg">No projects found</p>
                    </div>`;
                uiStateContainer.style.display = 'flex';
                return;
            }

            uiStateContainer.style.display = 'none';

            filteredData.forEach(item => {
                const isFav = favorites.includes(item.id);
                const heartClass = isFav ? "fa-solid text-rose-500 drop-shadow-[0_0_8px_rgba(244,63,94,0.6)]" : "fa-regular text-slate-500";
                
                // New Card Template
                const cardHTML = `
                    <a href="${item.link}" target="_self" class="group relative block bg-slate-800/80 backdrop-blur-sm rounded-2xl p-6 border border-slate-700 hover:-translate-y-2 hover:border-cyan-500 transition-all duration-300 shadow-lg hover:shadow-[0_10px_30px_-10px_rgba(6,182,212,0.3)] overflow-hidden">
                        
                        <div class="absolute top-0 left-0 w-full h-1 bg-gradient-to-r from-cyan-500 to-blue-500 transform origin-left scale-x-0 group-hover:scale-x-100 transition-transform duration-500"></div>

                        <div class="flex justify-between items-start mb-5 relative z-10">
                            <span class="bg-slate-900 text-cyan-400 text-xs font-bold px-3 py-1.5 rounded-md border border-slate-700 shadow-inner">
                                <i class="fa-solid fa-hashtag mr-1 opacity-50"></i>${item.id}
                            </span>
                            <button onclick="toggleFavorite(event, '${item.id}')" class="w-8 h-8 flex items-center justify-center bg-slate-900/50 rounded-full hover:bg-slate-700 transition-colors border border-transparent hover:border-slate-600 active:scale-90">
                                <i class="${heartClass} fa-heart"></i>
                            </button>
                        </div>
                        
                        <h2 class="text-2xl font-bold text-white mb-3 group-hover:text-cyan-400 transition-colors relative z-10">${item.title}</h2>
                        
                        <div class="space-y-2 mb-6 relative z-10">
                            <div class="flex items-center text-sm text-slate-400">
                                <div class="w-6 flex justify-center"><i class="fa-solid fa-layer-group text-slate-500"></i></div>
                                <span class="ml-2">${item.type}</span>
                            </div>
                            <div class="flex items-center text-sm text-slate-400">
                                <div class="w-6 flex justify-center"><i class="fa-solid fa-calendar-day text-slate-500"></i></div>
                                <span class="ml-2">${item.date}</span>
                            </div>
                        </div>
                        
                        <div class="w-full bg-transparent border-2 border-slate-600 text-slate-300 text-center py-2.5 rounded-xl font-semibold group-hover:border-cyan-500 group-hover:bg-cyan-500 group-hover:text-slate-900 transition-all duration-300 relative z-10 flex justify-center items-center gap-2">
                            Open Project <i class="fa-solid fa-arrow-right-long transform group-hover:translate-x-1 transition-transform"></i>
                        </div>
                    </a>`;
                cardContainer.innerHTML += cardHTML;
            });
        }

        // Toggle Favorites
        window.toggleFavorite = function(e, id) {
            e.preventDefault();
            e.stopPropagation();
            if (favorites.includes(id)) {
                favorites = favorites.filter(favId => favId !== id);
            } else {
                favorites.push(id);
            }
            localStorage.setItem('chakradham_favs', JSON.stringify(favorites));
            renderCards(document.getElementById('searchInput').value);
        };

        // Initialize App
        async function initApp() {
            const [fetchedAds, fetchedCards] = await Promise.all([
                fetchTextData('ads.txt', 'ads'),
                fetchTextData('data.txt', 'cards')
            ]);

            if (fetchedCards) {
                appData = fetchedCards;
                renderCards();
            }
            if (fetchedAds) {
                adsData = fetchedAds;
                startAds();
            }
        }

        // Search Listener
        document.getElementById('searchInput').addEventListener('input', (e) => {
            renderCards(e.target.value);
        });

        // Start App
        initApp();
    </script>
